# 基数排序

## LSD
``` cpp
int getBits(vector<int> arr) {
    int maxn = *max_element(arr.begin(), arr.end());
    int bits = 0;
    while (maxn > 0) {
        bits++;
        maxn /= 10;
    }
    return bits;
}

void radixSort(vector<int>& arr) {
    int bits = getBits(arr);
    int n = arr.size();
    int base = 10;

    vector<int> cnts(10);
    vector<int> help(n, 0);
    
    for (int offset = 1; bits > 0; offset *= base, bits--) {
        cnts.assign(10, 0);
        for (int i = 0; i < n; i++) {
            cnts[(arr[i] / offset) % base]++;
        }
        for (int i = 1; i < base; i++) {
            cnts[i] = cnts[i] + cnts[i - 1];
        }
        for (int i = n - 1; i >= 0; i--) {
            help[--cnts[(arr[i] / offset) % base]] = arr[i];        
        }
        for (int i = 0; i < n; i++) {
            arr[i] = help[i];
        }
    }
}
```
