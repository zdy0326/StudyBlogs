# 最小生成树

## Kruscal算法 (有权无向图)

```cpp
int find(vector<int> &f, int x) {
    if (f[x] == x) return x;
    return f[x] = find(f, f[x]);
}

void uniset(vector<int> &f, int x, int y) {
    int fx = find(f, x), fy = find(f, y);
    if (fx != fy) f[fy] = fx;
}

int main() {
    int n, m;
    cin >> n >> m;

    vector<int> f(n + 1);
    for (int i = 1; i <= n; i++) f[i] = i;

    vector<array<int, 3>> g;
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        g.push_back({u, v, w});
    }

    sort(g.begin(), g.end(),
         [](auto &a, auto &b) { return a[2] < b[2]; });

    long long ans = 0;
    int cnt = 0;

    for (auto &p : g) {
        if (find(f, p[0]) != find(f, p[1])) {
            uniset(f, p[0], p[1]);
            ans += p[2];
            cnt++;
        }
    }

    if (cnt != n - 1)
        cout << "orz";
    else
        cout << ans;

    return 0;
}


```

## Prim算法 (有权无向图)

```cpp
const int inf = 1e9 + 7;

int prim(int n, vector<vector<pair<int, int>>>& adj) {
    vector<int> dist(n, inf);
    vector<bool> used(n, false);
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;

    dist[0] = 0;
    pq.push({0, 0});
    
    int res = 0;

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (used[u]) continue;
        used[u] = true;
        res += d;

        for (auto [v, w] : adj[u]) {
            if (!used[v] && w < dist[v]) {
                dist[v] = w;
                pq.push({w, v});
            }
        }
    }

    for (bool b : used) {
        if (!b) return -1;
    }
    return res;
}
```
